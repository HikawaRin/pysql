CREATE TABLE IF NOT EXISTS SCSchema(
   ID INT UNSIGNED AUTO_INCREMENT,
   ParentID INT UNSIGNED,
   ElementName VARCHAR(100) NOT NULL,
   NodeType VARCHAR(100) NOT NULL,
   DataType VARCHAR(100),
   PreSibID INT UNSIGNED,
   Restriction VARCHAR(100),
   Restriction_value VARCHAR(100),
   PRIMARY KEY (ID)
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

TRUNCATE TABLE '表名'

CREATE TABLE IF NOT EXISTS Model(
   ID INT UNSIGNED AUTO_INCREMENT,
   ModelName VARCHAR(100) NOT NULL,
   PRIMARY KEY (ID)
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE IF NOT EXISTS SCData(
   ModelID INT UNSIGNED,
   SchemaID INT UNSIGNED,
   location INT UNSIGNED,
   Data VARCHAR(100),
   FOREIGN KEY (ModelID) REFERENCES Model(ID),
   FOREIGN KEY (SchemaID) REFERENCES SCSchemautf8(ID)   
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE IF NOT EXISTS UnitSchema(
   ID INT UNSIGNED AUTO_INCREMENT,
   ParentID INT UNSIGNED,
   ElementName VARCHAR(100) NOT NULL,
   OrderID INT UNSIGNED,
   NodeType VARCHAR(100),
   PRIMARY KEY (ID)
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE IF NOT EXISTS UnitData(
   ModelID INT UNSIGNED,
   SchemaID INT UNSIGNED,
   location VARCHAR(100),
   Data VARCHAR(100),
   FOREIGN KEY (ModelID) REFERENCES Model(ID),
   FOREIGN KEY (SchemaID) REFERENCES UnitSchema(ID)   
)ENGINE=InnoDB DEFAULT CHARSET=utf8;
